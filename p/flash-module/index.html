void SaveParam(void)
{
  /*保存参数至FLASH*/
  Store_Data[1] = GY_Offset;
  Store_Data[2] = *((uint16_t *)&AngleAcc_Offset);
  Store_Data[3] = *((uint16_t *)&AngleAcc_Offset + 1);
  Store_Data[4] = SpeedLevel;
  Store_Save();
}

void LoadParam(void)
{
  /*从FLASH加载参数*/
  uint32_t Temp;
  GY_Offset = Store_Data[1];
  Temp = (Store_Data[3] << 16) | Store_Data[2];
  AngleAcc_Offset = *(float *)&Temp;
  SpeedLevel = Store_Data[4];
}

  上面所展示的两个函数，就是平衡车进行保存陀螺仪Y轴——GY的偏移值以及由加速度计算出来的角度——AngleAcc_Offset的偏移值，还有小车的速度等级的两个相关函数。
  其中主要讲一下float类型的AngleAcc_Offset的存储策略！我们知道float类型的变量占用4个字节，而当前Store_Data[]数组的每个元素数据类型是int16_t类型的，存储的是2个字节的，所以就应当进行数据类型转换。
  首先，以上使用的逻辑是先取地址，再转化地址的类型，后再对转化后的地址进行取值操作。首先，对AngleAcc_Offset进行取地址，此时的地址默认是其首字节的内存地址，这个时候的地址是按4个字节为一组的，也就是这个地址下存储的是4个字节的数据，每次读取都是直接读出4个字节，且(&AngleAcc_Offset+1)的操作，是将该地址移动4个字节的数据宽度，而(* (&AngleAcc_Offset+1))就是取下个地址所存储的4个字节。但因为我们进行存储数据所使用的数组是int16_t类型的，所以要将变量AngleAcc_Offset拆分成前两个字节和后两个字节存储，所以，这里将他的地址强转为(int16_t * )，这个强转后的地址仍然是变量AngleAcc_Offset的首元素地址，但是，这次强转后，这个地址所能存储的字节大小变为了2个字节，将位置((uint16_t * )&AngleAcc_Offset + 1)，也就是将但当前地址移动两个字节数据宽度，此时取值后的的数据便为该变量的后两个字节数据，这样也就达成了我们的目的，将该变量拆分后分别存储起来，后续要使用时，再去他这两两字节拼接为4个字节的初始值!
  对于函数LoadParam(),就是将存储后的数据进行拼接得到原始数据。代码是使用中间值uint32_t Temp进行存储拼接后的值，因为进行存储时使用整数类型存储时，所以拼接完的数也是整数类型。但是变量AngleAcc_Offset是浮点数类型，他的编码格式和整数大大不同1，如果拼接后直接赋值给AngleAcc_Offset，就会导致最终得到的数据是错误的。所以,代码先对Temp取地址，然后转化为(float*) 类型的地址，然后再取值，就读取了正确的浮点数。
  而也许有人会问，为什么不直接(float)Temp？那是因为如果对Temp直接强转为float类型的数据时，Temp的数值其实已将是以整数进行转换而得出来的整数数据，这里进行float强转时，也只是在这个以解析成的整数的基础上，加上小数点和小数位0而已，这时解析的数据仍然是错误的。而取地址后将该地址转换为float类型的指针，然后再进行取值时，就是在告诉编译器说这块地址存储的是float类型的数据，进行读取时要以按浮点数的规则读，这时候读取到的数据才是正确的！

关于FLASH题外话

  接下来，这里要开展关于FLASH的题外话了，感兴趣的可以继续看看。
  模糊上来讲，单片机的内存可以分为 FLASH 区和 SRAM 区。FLASH 存储的数据掉电不丢失，我们通过编译器编译烧录进单片机的程序代码、常量都会存放在 FLASH；普通变量、运行临时参数存放在 SRAM 区，SRAM 通电期间可持续保存数据，一旦断电，数据会全部丢失。在全局变量前添加const关键字，该常量会存放至 FLASH，实现掉电数据不丢失。 举个例子：全局定义uint8_t A = 7;，变量 A 的内存空间分配在 SRAM 区，链接器把初始值7放到 FLASH 的.data段，记录该数值的 FLASH 源地址；同时记录变量A的 SRAM 目标地址；单片机上电启动时，启动代码会自动把 FLASH 里的数值 7 拷贝到 SRAM 中 A 对应的地址。 而全局定义const uint8_t A = 7;，变量对应的存储空间和数值 7 全部存放在 FLASH 只读段，不占用 SRAM，程序运行时直接读取 FLASH 数据，无需上电拷贝加载。
  而在HAL库与标准库的对比提过，HAL库的代码比较冗杂,这就导致了使用HAL库进行编写的程序，FLASH的使用率就会比较高些。例如，在我进行Debug期间，FLASH的使用率高达95%以上，也就是64KB的FLASH内存已经使用了61KB左右了，如果在使用下去，内存就会发生溢出，可能会导致一些奇奇怪怪的BUG。而且，我们已经将最后1KB字节拿来存储数据了，实际可用内存已经不足2KB.所以我去对代码进行了优化删减，把一些不必要的代码删去，也把OLED模块的一些没用到的数据删去了(因为z这些数据加了const)。整改后FLASH的使用减少至55KB。但是，我又去对比一下标准库的使用情况，发现其仅使用了不到35KB.但是我的代码已经在做到了最简，我实在想不出哪些可以剔除。后面问了一下AI，发现是CMake开启了浮点数，导致了链接器把完整的软件浮点 printf 库（_printf_float）塞进 FLASH，用于支持 sprintf("%.2f", …) 这类浮点格式化输出。这个库本身就占 ~10KB（实测6KB）。所以，我将浮点数打印关了，但是浮点数打印又是当前工程需求的，所以，需要另辟蹊径！

/* Float_Printf showSign 参数宏定义 */
#define FLOAT_SIGN_OnlyNEGATIVE    0   /* 仅负数显示-，正数不显示符号 */
#define FLOAT_SIGN_ALWAYS          1   /* 始终显示 +/- */
#define FLOAT_SIGN_OnlyPOSITIVE    2   /* 仅正数显示+，负数不显示符号 */

static char Float_Buf[FLOAT_BUF_COUNT][FLOAT_BUF_SIZE];
static uint8_t Float_BufIdx = 0;


/**
 * @brief  浮点数格式化为字符串（替代 sprintf 的 %f，不依赖 _printf_float）
 * @param  value    浮点数值
 * @param  width    总显示宽度（含符号、整数部分、小数点、小数部分）
 * @param  decimals 小数位数（0~5）
 * @param  showSign FLOAT_SIGN_OnlyNEGATIVE=仅负数显示-，FLOAT_SIGN_ALWAYS=始终显示+/-，FLOAT_SIGN_OnlyPOSITIVE=仅正数显示+
 * @return 格式化后的字符串指针（旋转缓冲区）
 */
char* Float_Printf(float value, int width, int decimals, uint8_t showSign)
{
  char* buf = Float_Buf[Float_BufIdx];
  Float_BufIdx = (Float_BufIdx + 1) % FLOAT_BUF_COUNT;

  int pos = 0;
  char sign = 0;

  /* 1. 处理符号 */
  if (value < 0.0f)
  {
    if (showSign != 2) sign = '-';  /* showSign=2时不显示负号，其他都显示 */
    value = -value;
  }
  else if (showSign >= 1)
  {
    sign = '+';  /* showSign=1或2时显示正号 */
  }

  /* 2. 拆分整数部分和小数部分 */
  int intPart = (int)value;
  float frac = value - (float)intPart;

  /* 3. 计算小数部分（0~5位），不用pow，用乘法循环 */
  int fracInt = 0;
  if (decimals > 0)
  {
    float multiplier = 1.0f;
    for (int i = 0; i < decimals; i++) multiplier *= 10.0f;
    fracInt = (int)(frac * multiplier);  /* 直接截断，不四舍五入 */
  }

  /* 4. 计算整数部分的位数 */
  int intDigits = 0;
  int temp = intPart;
  if (temp == 0)
  {
    intDigits = 1;
  }
  else
  {
    while (temp > 0)
    {
      intDigits++;
      temp /= 10;
    }
  }

  /* 5. 计算总所需长度：符号 + 整数位 + 小数点 + 小数位 */
  int totalLen = (sign ? 1 : 0) + intDigits + (decimals > 0 ? 1 : 0) + decimals;

  /* 6. 计算需要补零的数量（符号占1位后，剩余宽度用于补零+整数+小数点+小数） */
  int padZero = 0;
  int contentLen = intDigits + (decimals > 0 ? 1 : 0) + decimals;
  if (width > (sign ? 1 : 0) + contentLen)
  {
    padZero = width - (sign ? 1 : 0) - contentLen;
  }

  /* 7. 从左到右构建字符串：符号 -> 补零 -> 整数部分 -> 小数点 -> 小数部分 */

  /* 符号（放在最前面） */
  if (sign)
  {
    buf[pos++] = sign;
  }

  /* 补零 */
  for (int i = 0; i < padZero; i++)
  {
    buf[pos++] = '0';
  }

  /* 整数部分（递归写入） */
  if (intPart == 0)
  {
    buf[pos++] = '0';
  }
  else
  {
    /* 先写高位到临时数组，再反转写入buf */
    char intBuf[12];
    int intLen = 0;
    temp = intPart;
    while (temp > 0)
    {
      intBuf[intLen++] = '0' + (temp % 10);
      temp /= 10;
    }
    /* 反转 */
    for (int i = intLen - 1; i >= 0; i--)
    {
      buf[pos++] = intBuf[i];
    }
  }

  /* 小数点 */
  if (decimals > 0)
  {
    buf[pos++] = '.';

    /* 小数部分（补零 + 数字） */
    int fracDigits = 1;
    int divisor = 1;
    for (int i = 1; i < decimals; i++)
    {
      divisor *= 10;
      fracDigits *= 10;
    }
    fracDigits *= 10;

    for (int i = 0; i < decimals; i++)
    {
      int digit = (fracInt / divisor) % 10;
      buf[pos++] = '0' + digit;
      divisor /= 10;
    }
  }

  buf[pos] = '\0';
  return buf;
}

  编译信息就会有FLASH和SRAM的使用信息，只是形式不同而已！

Code=28894：程序代码（.text，存 Flash）
RO-data=5330：const 常量、字符串只读数据（.rodata，存 Flash）
RW-data=408：带初始值全局变量的初始值（.data，烧录在 Flash，上电拷贝到 SRAM）
ZI-data=4064：零初始化 / 未赋值全局变量、栈堆（只占 SRAM，不占用 Flash）

总 Flash = Code + RO-data + RW-data


.text：程序代码、函数指令、const 常量 / 字符串（对应 Keil 的 Code + RO-data），全部存 Flash。
.data：带初始值的全局变量（对应 Keil RW-data），初始值存 Flash，上电启动代码拷贝到 SRAM。
.bss：未初始化 / 零初始化全局变量、堆、栈，只占用 SRAM，不占 Flash。
Flash 占用 = .text + .data
